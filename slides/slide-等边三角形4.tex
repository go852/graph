\usetikzlibrary{intersections,through}
\begin{document}
\begin{tikzpicture}
    % 定义点 A 和 B
    \coordinate (A) at (0,0);
    \coordinate (B) at (5,0);
    
    % 绘制原始线段 AB
    \draw[thick] (A) --++ (60:5) coordinate (C) -- (B) -- cycle;
    \node at (A) [below left] {$A$};
    \node at (B) [below right] {$B$};
    \node at (C) [above] {$C$}; 
\end{tikzpicture}
\end{document}
